Regras do Firestore para os apps gerados

Cole isto no Console do Firebase do projeto forja-ia → Firestore Database → Regras. Isso permite que cada app publicado tenha login próprio e salve dados isolados por usuário, sem expor dados de outros apps ou projetos.

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // --- Coleções do PRÓPRIO Forja (builder, projetos, login do Josy) ---
    match /projects/{projectId} {
      allow read, update, delete: if request.auth != null
                                   && request.auth.uid == resource.data.userId;
      allow create: if request.auth != null;
    }
    match /publicApps/{appId} {
      allow read: if true;
      allow write: if request.auth != null;
    }

    // --- Dados dos APPS GERADOS pela IA (qualquer coleção começando com "app_") ---
    // Cada usuário final só lê/escreve dentro do próprio uid.
    match /{appCollection}/{uid}/{document=**} {
      allow read, write: if appCollection.matches('app_.*')
                          && request.auth != null
                          && request.auth.uid == uid;
    }
  }
}
